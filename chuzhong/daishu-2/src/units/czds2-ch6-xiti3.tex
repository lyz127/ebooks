\xiaoti{地球的半径 $R = 6.4 \times 10^6$ 米。球的体积公式是 $V = \dfrac{4}{3} \pi R^3$。
    求地球的体积（$\pi$ 取 3.14，结果保留两个有效数字）。
}

\xiaoti{计算：}
\begin{xiaoxiaotis}

    \begin{tblr}{columns={18em, colsep=0pt}, column{2}={20em}}

        \xxt{$2x \cdot \left(x^2 - \dfrac{1}{2}x + 1\right)$；} & \xxt{$5ab \cdot (2a - b + 0.2)$；} \\
        \xxt{$(-2ab^2)^2 \cdot (3a^2b - 2ab - 4b^3)$；} & \xxt{$\left(\dfrac{3}{4}x^2y - \dfrac{1}{2}xy^2 - \dfrac{5}{6}y^3\right) \cdot (-4xy^2)$。}
