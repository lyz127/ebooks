\xiaoti{解下列不等式， 并把它们的解集在数轴上表示出来：}
\begin{xiaoxiaotis}

    \begin{tblr}{columns={18em, colsep=0pt}}
        \xxt{$2(x - 3) > 4$；} & \xxt{$2x - 3 \leqslant 5(x - 3)$；} \\
        \xxt{$\dfrac{1}{5}(x - 2) \leqslant x - \dfrac{2}{5}$；} & \xxt{$\dfrac{x}{3} - \dfrac{x - 1}{2} < 1$。}
    \end{tblr}

\end{xiaoxiaotis}

\xiaoti{下面各题的解法对不对，为什么？}
\begin{xiaoxiaotis}

    \xxt{$-x = 6$，两边都乘以 $-1$，得 $x = -6$；}

    \xxt{$-x > 6$，两边都乘以 $-1$，得 $x > -6$；}

    \xxt{$-x \leqslant 6$，两边都乘以 $-1$，得 $x \leqslant -6$。}

\end{xiaoxiaotis}

\xiaoti{下面各题的解法对不对，为什么？}
\begin{xiaoxiaotis}

    \begin{tblr}{columns={18em, colsep=0pt}}
        \xxt{$7x + 5 > 8x + 6 \juhao $ \\
                \hspace*{.5em} $\begin{aligned}[t]
                    7x - 8x &> 6 - 5 , \\
                    -x &> 1 , \\
                    x &> -1 \juhao
                \end{aligned}$}
            &\xxt{$6x - 3 < 4x - 4 \juhao$ \\
                \hspace*{.5em} $\begin{aligned}[t]
                    6x -4x &< -4 + 3, \\
                    2x &< -1, \\
                    x &> \dfrac{1}{2} \juhao
                \end{aligned}$}
    \end{tblr}

\end{xiaoxiaotis}

\begin{withstar}
\xiaoti{用小于号 “$<$” 、等号 “$=$” 或大于号 “$>$” 填空：}
\begin{xiaoxiaotis}

    \xxt{如果 $a < b$，那么 $a - b \xhx 0$；}

    \xxt{如果 $a = b$，那么 $a - b \xhx 0$；}

    \xxt{如果 $a > b$，那么 $a - b \xhx 0$。}

\end{xiaoxiaotis}

\xiaoti{用小于号 “$<$” 、等号 “$=$” 或大于号 “$>$” 填空：}
\begin{xiaoxiaotis}

    \xxt{如果 $a - b < 0$，那么 $a \xhx b$；}

    \xxt{如果 $a - b = 0$，那么 $a \xhx b$；}

    \xxt{如果 $a - b > 0$，那么 $a \xhx b$。}

\end{xiaoxiaotis}

\xiaoti{用小于号 “$<$” 或大于号 “$>$” 填空：}

    \xxt{当 $a >0$， $b \xhx 0$， $ab > 0$；}

    \xxt{当 $a >0$， $b \xhx 0$， $ab < 0$。}

\end{xiaoxiaotis}
